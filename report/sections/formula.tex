\section{Mathematical Model}

\subsection{Sets and Indices}

\begin{itemize}
  \item $p \in P$ -- set of patients (surgeries)
  \item $P_0 = P \cup \{0\}$ -- patients plus dummy depot node (session start)
  \item $h \in H$ -- set of available OR sessions
  \item $s \in S$ -- set of scenarios
  \item $q \in Q = \{\text{CHI},\text{ORT},\text{KNO}\}$ -- set of surgical
        specialties
\end{itemize}

\subsection{Parameters}

\begin{itemize}
  \item $d_{ps}$ -- surgery duration of patient $p$ in scenario $s$
  \item $\pi_s$ -- probability of scenario $s$,
        where $\sum_{s \in S} \pi_s = 1$
  \item $c$ -- changeover time
  \item $t^{\text{start}}$ -- common session start time (identical for all ORs)
  \item $t^{\text{close}}$ -- common session closing time (identical for all
        ORs)
  \item $q_{pq} \in \{0,1\}$ -- 1 if patient $p$ belongs to specialty $q$
  \item $M$ -- sufficiently large constant (big-$M$)
  \item $\beta_1, \beta_2, \beta_3, \beta_4$ -- weights for waiting time, idle
        time, overtime, and specialty mismatch, respectively
\end{itemize}

\subsection{Decision Variables}

\subsubsection{First-Stage Variables}
Decided before scenario realisation:
\begin{itemize}
  \item $A_p \geq 0$ -- appointment time assigned to patient $p$
  \item $X_{ijh} \in \{0,1\}$ -- 1 if patient $j$ is scheduled directly after
        patient $i$ in session $h$
  \item $Y_{ph} \in \{0,1\}$ -- 1 if patient $p$ is assigned to session $h$
  \item $Z_h \in \{0,1\}$ -- 1 if OR session $h$ is opened
  \item $V_{hq} \in \{0,1\}$ -- 1 if session $h$ is assigned specialty $q$
  \item $D_{hq} \in \mathbb{Z}_{\geq 0}$ -- number of patients in session $h$ with specialty $q$
        that are not placed in a session with their specialty (violation count)
  \item $U_{ph} \in \mathbb{Z}_{\geq 0}$ -- position of patient $p$ in session
        $h$ (MTZ auxiliary variable)
\end{itemize}

\subsubsection{Second-Stage Variables}
Evaluated per scenario $s$:
\begin{itemize}
  \item $S_{ps} \geq 0$ -- start time of patient $p$ under
        scenario $s$
  \item $W_{ps} \geq 0$ -- waiting time of patient $p$ under scenario $s$
  \item $I_{hs} \geq 0$ -- idle time of session $h$ under scenario $s$
  \item $O_{hs} \geq 0$ -- overtime of session $h$ under scenario $s$
\end{itemize}

\subsection{Objective Function}

The objective minimises the expected weighted sum of patient waiting time, OR
idle time, and OR overtime, together with a scenario-independent penalty for
specialty violations:
%
\begin{align}
  \min \quad
    & \sum_{s \in S} \pi_s \Bigl[
        \beta_1 \sum_{p \in P} W_{ps}
      + \beta_2 \sum_{h \in H} I_{hs}
      + \beta_3 \sum_{h \in H} O_{hs}
      \Bigr] \nonumber \\
    & + \; \beta_4 \sum_{h \in H} \sum_{q \in Q} D_{hq}
  \label{eq:obj}
\end{align}

The mismatch penalty is scenario-independent because specialty and assignment
decisions are first-stage.

\subsection{Constraints}

\subsubsection{Appointment Time Bounds}

Each appointment must fall within the OR session window:
%
\begin{equation}
  t^{\text{start}} \leq A_p \leq t^{\text{close}}
  \qquad \forall\, p \in P
  \label{eq:appt_bounds}
\end{equation}

\subsubsection{Assignment and Session Opening}

Each patient must be assigned to exactly one session:
%
\begin{equation}
  \sum_{h \in H} Y_{ph} = 1
  \qquad \forall\, p \in P
  \label{eq:assign}
\end{equation}

A patient can only be assigned to an opened session:
%
\begin{equation}
  Y_{ph} \leq Z_h
  \qquad \forall\, p \in P,\; h \in H
  \label{eq:open}
\end{equation}

\subsubsection{Specialty Assignment}

Each opened session is assigned exactly one specialty; closed sessions receive
none:
%
\begin{equation}
  \sum_{q \in Q} V_{hq} = Z_h
  \qquad \forall\, h \in H
  \label{eq:spec}
\end{equation}

\subsubsection{Specialty Violation Counting}

The integer variable $D_{hq}$ counts the number of patients in session $h$
whose specialty does not match $q$. For each session $h$ and specialty $q$,
the number of assigned patients belonging to specialty $q$ cannot exceed the
session capacity unless a violation is recorded:
%
\begin{equation}
  \sum_{p \in P} Y_{ph} \cdot q_{pq}
  \leq M \cdot V_{hq} + D_{hq}
  \qquad \forall\, h \in H,\; q \in Q
  \label{eq:viol}
\end{equation}

When session $h$ is assigned specialty $q$ (i.e.\ $V_{hq} = 1$), the big-$M$
term is active and $D_{hq} = 0$ is optimal under minimisation provided all
assigned patients match. When $V_{hq} = 0$, the big-$M$ term vanishes and
$D_{hq}$ must be at least as large as the number of patients of specialty $q$
placed in session $h$, recording each as a violation.

\subsubsection{Routing --- Flow Conservation}

The sequencing variables $X_{ijh}$ define a single route per open session,
mirroring a vehicle routing formulation:
%
\begin{align}
  \sum_{p \in P} X_{0ph} &= Z_h
  \qquad \forall\, h \in H
  \label{eq:depot} \\[2pt]
  \sum_{\substack{p' \in P_0 \\ p' \neq p}} X_{pp'h} &= Y_{ph}
  \qquad \forall\, p \in P,\; h \in H
  \label{eq:flow_out} \\[2pt]
  \sum_{\substack{p \in P_0 \\ p \neq p'}} X_{pp'jh} &= Y_{p'h}
  \qquad \forall\, p' \in P,\; h \in H
  \label{eq:flow_in}
\end{align}

\subsubsection{Timing --- Start Time Propagation}

If patient $j$ immediately follows patient $i$ in session $h$, the start time
of $j$ must respect $i$'s realised duration and the changeover time:
%
\begin{align}
  S_{p's} \geq\; & S_{ps} + d_{ps} + c
                   - M\bigl(1 - X_{pp'h}\bigr) \nonumber \\
                 & \forall\, p,p' \in P,\; p \neq p',\; h \in H,\; s \in S
  \label{eq:time_seq}
\end{align}

The first patient in a session starts no earlier than the session opening plus
its changeover from the depot:
%
\begin{align}
  S_{ps} \geq\; & t^{\text{start}}
                   - M\bigl(1 - X_{0ph}\bigr) \nonumber \\
                 & \forall\, p \in P,\; h \in H,\; s \in S
  \label{eq:time_first}
\end{align}

\subsubsection{Session Lower Bounds}

A patient's surgery cannot start before the session opens or before the
patient's appointment time:
%
\begin{align}
  S_{ps} &\geq A_p - M(1 - Y_{ph})
  \qquad \forall\, p \in P,\; h \in H,\; s \in S
  \label{eq:lb_appt}
\end{align}


\subsubsection{Waiting Time}

Patient waiting time is the difference between actual start time and the
decision-variable appointment time:
%
\begin{equation}
  W_{ps} \geq S_{ps} - A_p
  \qquad \forall\, p \in P,\; s \in S
  \label{eq:wait}
\end{equation}

\subsubsection{Overtime}

Overtime in session $h$ under scenario $s$ is the amount by which the last
patient's finish time exceeds the closing time:
%

\begin{align}
  O_{hs} \geq\; &
                    S_{ps} + d_{ps}
                  - t^{\text{close}}-M(1- Y_{ph}) \nonumber \\
                & \forall\, p \in P,\;\forall\, h \in H,\; s \in S
  \label{eq:overtime}
\end{align}

\subsubsection{Idle Time}

Idle time is the session window minus total utilisation, corrected for
overtime. Multiplying by $Z_h$ ensures closed sessions incur no idle time:
%
\begin{align}
  I_{hs} \geq\; & \bigl(t^{\text{close}} - t^{\text{start}}\bigr) Z_h
                  - \sum_{p \in P} d_{ps}\, Y_{ph}
                  - c \Bigl(\sum_{p \in P} Y_{ph} - Z_h\Bigr)
                  + O_{hs} \nonumber \\
                & \forall\, h \in H,\; s \in S
  \label{eq:idle}
\end{align}

\subsubsection{Sub-Tour Elimination}

Miller--Tucker--Zemlin (MTZ) constraints prevent disconnected cycles within a
session:
%
\begin{align}
  U_{ph} - U_{p'h} + |P|\, X_{pp'h} \leq |P| - 1 \nonumber \\
  \forall\, p,p' \in P,\; p \neq p',\; h \in H
  \label{eq:mtz}
\end{align}

\subsubsection{Domain}

\begin{align}
  & t^{\text{start}} \leq A_p \leq t^{\text{close}}
  \qquad \forall\, p \in P
  \label{eq:dom_appt} \\[2pt]
  & X_{h},\; Y_{ph},\; Z_h,\; V_{hq} \in \{0,1\}
  \label{eq:dom_bin} \\[2pt]
  & D_{hq} \in \mathbb{Z}_{\geq 0}
  \qquad \forall\, h \in H,\; q \in Q
  \label{eq:dom_viol} \\[2pt]
  & S_{ps},\; W_{ps},\; I_{hs},\; O_{hs} \geq 0
  \label{eq:dom_cont} \\[2pt]
  & U_{ph} \in \mathbb{Z}_{\geq 0}
  \label{eq:dom_mtz}
\end{align}
